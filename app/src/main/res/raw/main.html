40           }
41       }
42   
43       //constructor
44       public MainAdapter(ArrayList<SubjectList> objects, Context context) {
45           this.objects = objects;

49       @Override
50       public MainViewHolder onCreateViewHolder(ViewGroup parent, int viewType) {
51   
52           View view = LayoutInflater.from(parent.getContext())
53                   .inflate(R.layout.main_card_view,parent,false);
54           return new MainViewHolder(view);
55       }
56   
57       @Override
58       public void onBindViewHolder(final MainViewHolder holder, int position) {
59   
60           final SubjectList list = objects.get(position);
61   
62           holder.textView.setText(list.getSubName());
63   
64           holder.textView.setOnClickListener(new View.OnClickListener() {
65               @Override
66               public void onClick(View v) {
67                   switch (holder.getAdapterPosition()){
68                       case 0:
69                           context.startActivity(new Intent(context, LocationActivity.class));
70                           break;
71                       case 1:
72                           context.startActivity(new Intent(context, EmailValidationActivity.class));
73                           break;
74   
75                   }
76               }
77           });

79       }
